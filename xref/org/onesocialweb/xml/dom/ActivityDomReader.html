43  	public ActivityDomReader() {
44  		this.factory = getActivityFactory();

49  	/*
50  	 * (non-Javadoc)
51  	 *
52  	 * @see
53  	 * org.onesocialweb.model.activity.ActivityDomReader#readEntry(org.w3c.dom
54  	 * .Element)
55  	 */
56  	public ActivityEntry readEntry(Element element) {
57  		ActivityEntry entry = factory.entry();
58  
59  		// The atom stuff
60  		entry.setId(DomHelper.getElementText(element, "id", Atom.NAMESPACE));
61  		entry.setTitle(DomHelper.getElementText(element, "title", Atom.NAMESPACE));
62  		entry.setPublished(parseDate(DomHelper.getElementText(element, "published", Atom.NAMESPACE)));
63  		entry.setUpdated(parseDate(DomHelper.getElementText(element, "updated", Atom.NAMESPACE)));
64  
65  		// Get the actor (ignore if more than one)
66  		Element actor = DomHelper.getElement(element, "actor", Activitystreams.NAMESPACE);
67  		if (actor != null) {
68  			entry.setActor(readActor(actor));
69  		}
70  
71  		// Get the verbs
72  		NodeList verbs = element.getElementsByTagNameNS(Activitystreams.NAMESPACE, "verb");
73  		for (int i = 0; i < verbs.getLength(); i++) {
74  			entry.addVerb(readVerb((Element) verbs.item(i)));
75  		}
76  
77  		// Get the objects
78  		NodeList objects = element.getElementsByTagNameNS(Activitystreams.NAMESPACE, "object");
79  		for (int i = 0; i < objects.getLength(); i++) {
80  			entry.addObject(readObject((Element) objects.item(i)));
81  		}
82  
83  		// Get the acl rules
84  		NodeList rules = element.getElementsByTagNameNS(Onesocialweb.NAMESPACE, "acl-rule");
85  		for (int i = 0; i < rules.getLength(); i++) {
86  			entry.addAclRule(aclDomReader.readRule((Element) rules.item(i)));
87  		}
88  
89  		// Get the reply-to
90  		NodeList replies = element.getElementsByTagNameNS(AtomThreading.NAMESPACE, AtomThreading.IN_REPLY_TO_ELEMENT);
91  		for(int i=0; i < replies.getLength(); i++) {
92  			AtomReplyTo replyTo= atomDomReader.readReplyTo((Element) replies.item(i));
93  			if (replyTo.getHref()!=null){
94  				entry.setParentId(readParentId(replyTo.getHref()));
95  				entry.setParentJID(readParentJID(replyTo.getHref()));
96  			}
97  			entry.addRecipient(replyTo);
98  		}
99  
100 		//Get the links...
101 		NodeList links = element.getElementsByTagNameNS(Atom.NAMESPACE, Atom.LINK_ELEMENT);
102 		for(int i=0; i < links.getLength(); i++) {
103 			AtomLink link= atomDomReader.readLink(((Element) links.item(i)));
104 			entry.addLink(link);
105 		}
106 
107 
108 		return entry;
109 	}

111 	public String readParentJID(String href)
112 	{
113 		if (href.length() == 0) {
114 			return null;
115 		}
116 
117 		int i = href.indexOf("?");
118 
119 		if (i > 0) {
120 			return href.substring(5, i);
121 		} else {
122 			return null;
123 		}       
124 	}
125 	
126 	public String readParentId(String href)
127 	{
128 		if (href.length() == 0) {
129 			return null;
130 		}
131 
132 		int i = href.indexOf("item=");
133 
134 		if (i > 0) {
135  			return href.substring(5+i, href.length());
136 		} else {
137 			return null;
138 		}
139 	}

141 	public String readActivityId(Element element) {
142 		return DomHelper.getElementAttribute(element, "id");
143 
144 	}

146 	protected ActivityActor readActor(Element element) {
147 		ActivityActor actor = factory.actor();
148 		actor.setUri(DomHelper.getElementText(element, "uri", Atom.NAMESPACE));
149 		actor.setName(DomHelper.getElementText(element, "name", Atom.NAMESPACE));
150 		actor.setEmail(DomHelper.getElementText(element, "email", Atom.NAMESPACE));
151 		return actor;
152 	}
153 
154 	protected ActivityVerb readVerb(Element element) {
155 		ActivityVerb verb = factory.verb();
156 		verb.setValue(element.getTextContent().trim());
157 		return verb;
158 	}
159 
160 	protected ActivityObject readObject(Element element) {
161 		ActivityObject object = factory.object();
162 
163 		object.setId(DomHelper.getElementText(element, "id", Atom.NAMESPACE));
164 		object.setType(DomHelper.getElementText(element, "object-type", Activitystreams.NAMESPACE));
165 		object.setTitle(DomHelper.getElementText(element, "title", Atom.NAMESPACE));
166 		object.setPublished(parseDate(DomHelper.getElementText(element, "published", Atom.NAMESPACE)));
167 		object.setUpdated(parseDate(DomHelper.getElementText(element, "updated", Atom.NAMESPACE)));
168 
169 		// content
170 		NodeList contents = element.getElementsByTagNameNS(Atom.NAMESPACE, "content");
171 		for (int i = 0; i < contents.getLength(); i++) {
172 			object.addContent(atomDomReader.readContent((Element) contents.item(i)));
173 		}
174 
175 		// links
176 		NodeList links = element.getElementsByTagNameNS(Atom.NAMESPACE, "link");
177 		for (int i = 0; i < links.getLength(); i++) {
178 			object.addLink(atomDomReader.readLink((Element) links.item(i)));
179 		}
180 
181 		return object;
182 	}
