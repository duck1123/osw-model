44  	public void toXml(Profile profile, StringBuffer buffer) {
45  		// Prepare for writing
46  		this.buffer = buffer;
47  
48  		// Open the entry tag
49  		startTag(VCard4.VCARD_ELEMENT);
50  		attribute("xmlns", VCard4.NAMESPACE);
51  		attribute("xmlns:osw", Onesocialweb.NAMESPACE);
52  		endOpen();
53  
54  			
55  
56  		// Dump the fields
57  		for (Field field : profile.getFields()) {
58  			if (field instanceof BirthdayField) {
59  				dump((BirthdayField) field);
60  			} else if (field instanceof FullNameField) {
61  				dump((FullNameField) field);
62  			} else if (field instanceof GenderField) {
63  				dump((GenderField) field);
64  			} else if (field instanceof NoteField) {
65  				dump((NoteField) field);
66  			} else if (field instanceof PhotoField) {
67  				dump((PhotoField) field);	
68  			} else if (field instanceof EmailField) {
69  				dump((EmailField) field);
70  			} else if (field instanceof NameField) {
71  				dump((NameField) field);
72  			} else if (field instanceof TelField) {
73  				dump((TelField) field);
74  			} else if (field instanceof TimeZoneField) {
75  				dump((TimeZoneField) field);
76  			} else if (field instanceof URLField) {
77  				dump((URLField) field);
78  			} 
79  		}
80  		// Close
81  		closeTag(VCard4.VCARD_ELEMENT);		
82  	}

84  	private void dump(BirthdayField field) {
85  		openTag(VCard4.BDAY_ELEMENT);
86  		
87  		BirthdayField.Type type= field.getType();
88  		if (type.equals(BirthdayField.Type.Date))
89  			text(VCard4.DATE_ELEMENT, field.getValue());
90  		if (type.equals(BirthdayField.Type.DateTime))
91  			text(VCard4.DATETIME_ELEMENT, field.getValue());
92  		if (type.equals(BirthdayField.Type.Time))
93  			text(VCard4.TIME_ELEMENT, field.getValue());
94  		
95  		dumpAclRules(field);
96  		closeTag(VCard4.BDAY_ELEMENT);
97  	}

232 	private void dumpAclRules(Field field) {
233 		if (field.hasAclRules()) {
234 			for (AclRule rule : field.getAclRules()) {
235 				openTag("osw:" + Onesocialweb.ACL_RULE_ELEMENT);
236 				if (rule.hasActions()) {
237 					for (AclAction action : rule.getActions()) {
238 						startTag("osw:" + Onesocialweb.ACL_ACTION_ELEMENT);
239 						attribute(Onesocialweb.PERMISSION_ATTRIBUTE, action.getPermission());
240 						endOpen();
241 						if (action.hasName())
242 							buffer.append(xmlEncode(action.getName()));
243 						closeTag("osw:" + Onesocialweb.ACL_ACTION_ELEMENT);
244 					}
245 				}
246 				if (rule.hasSubjects()) {
247 					for (AclSubject subject : rule.getSubjects()) {
248 						startTag("osw:" + Onesocialweb.ACL_SUBJECT_ELEMENT);
249 						attribute(Onesocialweb.TYPE_ATTRIBUTE, subject.getType());
250 						endOpen();
251 						if (subject.hasName())
252 							buffer.append(xmlEncode(subject.getName()));
253 						closeTag("osw:" + Onesocialweb.ACL_SUBJECT_ELEMENT);
254 					}
255 				}
256 				closeTag("osw:" + Onesocialweb.ACL_RULE_ELEMENT);
257 			}
258 		}
